export function day_as_word(day_number) {
  switch (day_number) {
    case 4: {
      return "Cuatro"
    }
  }
}

export function get_day(date) {
  const _date =
    typeof date === "string" ? new Date(date) : date
  return _date.getDay()
}

export function get_month(date) {
  const _date =
    typeof date === "string" ? new Date(date) : date
  return _date.toLocaleString("es-AR", { month: "long" }).toLocaleUpperCase()
}

export function get_year(date) {
  const _date =
    typeof date === "string" ? new Date(date) : date
  return _date.getFullYear()
}
